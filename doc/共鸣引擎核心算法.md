/**
 * DRTA (Dynamic Resonance Tensor Algorithm) v3.0 - Robust Edition
 * * 核心改进：
 * 1. 引入 Epsilon 因子防止除零错误。
 * 2. 数值钳制 (Clamping) 防止指数爆炸。
 * 3. 维度验证与异常捕获。
 */

export class LemeoneResonanceEngine {
  private static readonly EPSILON = 1e-10; // 极小量，用于数值稳定性
  private static readonly MAX_ENTROPY = 20; // 限制最大熵值，防止指数收益归零

  /**
   * 安全的余弦相似度计算
   * 预防：零向量导致的 NaN
   */
  public static calculateCosineSimilarity(vecA: number[], vecB: number[]): number {
    if (vecA.length !== vecB.length || vecA.length === 0) return 0;

    let dotProduct = 0;
    let normA = 0;
    let normB = 0;

    for (let i = 0; i < vecA.length; i++) {
      dotProduct += vecA[i] * vecB[i];
      normA += vecA[i] * vecA[i];
      normB += vecB[i] * vecB[i];
    }

    const magnitude = Math.sqrt(normA) * Math.sqrt(normB);

    // 防御：如果任何一个向量模长接近 0，返回 0 而不是 NaN
    if (magnitude < this.EPSILON) return 0;

    // 确保结果在 [-1, 1] 之间（浮点数精度可能导致微小越界）
    return Math.max(-1, Math.min(1, dotProduct / magnitude));
  }

  /**
   * 安全的指数衰减计算
   * 预防：熵值过大导致产出瞬间降为 0 或计算溢出
   */
  public static calculateEntropyDecay(entropy: number, lambda: number = 0.1): number {
    // 钳制熵值范围
    const clampedEntropy = Math.max(0, Math.min(this.MAX_ENTROPY, entropy));
    // 计算 e^(-λE)
    return Math.exp(-lambda * clampedEntropy);
  }

  /**
   * 核心产出演算
   * 实现原理：O = ||V_state|| * Cos(θ) * Decay
   */
  public static calculateOutput(
    founderVector: number[],
    staffMatrices: number[][], // 团队对向量的干涉
    marketVector: number[]
  ): { 
    output: number; 
    resonance: number; 
    magnitude: number; 
    entropy: number;
    isValid: boolean;
  } {
    try {
      // 1. 向量合成 (Vector Synthesis)
      // 创始人向量作为基底
      let stateVector = [...founderVector];

      // 叠加团队干涉条纹 (简化为向量累加，实际可升级为矩阵乘法)
      staffMatrices.forEach(matrix => {
        stateVector = stateVector.map((val, i) => val + (matrix[i] || 0));
      });

      // 2. 模长计算 (Magnitude)
      const magnitude = Math.sqrt(stateVector.reduce((sum, v) => sum + v * v, 0));

      // 3. 共鸣度计算 (Resonance)
      const resonance = this.calculateCosineSimilarity(stateVector, marketVector);

      // 4. 熵计算 (基于技术债和团队规模)
      // 熵 E = (团队数 * 0.5) + (技术债 * 1.2)
      const entropy = (staffMatrices.length * 0.5) + 1.0; // 这里的 1.0 为示例基础熵

      // 5. 最终计算
      const decay = this.calculateEntropyDecay(entropy);
      
      // 最终产出公式，且确保不为负
      const output = Math.max(0, magnitude * resonance * decay);

      return {
        output: isNaN(output) ? 0 : output,
        resonance,
        magnitude,
        entropy,
        isValid: !isNaN(output)
      };
    } catch (error) {
      console.error("[ENGINE_CRITICAL] 数值演算发生崩溃:", error);
      return { output: 0, resonance: 0, magnitude: 0, entropy: 0, isValid: false };
    }
  }

  /**
   * 市场漂移计算 (受限随机游走)
   * 预防：市场向量漂移至无穷远导致系统失效
   */
  public static nextMarketDrift(currentVector: number[]): number[] {
    return currentVector.map(v => {
      const drift = (Math.random() - 0.5) * 0.1; // 限制单次漂移量
      const next = v + drift;
      // 钳制在 [0, 1] 空间，防止市场需求变成负数或无限大
      return Math.max(0.01, Math.min(1, next));
    });
  }
}
